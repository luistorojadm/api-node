//Creacion del Proyecto

* Inicializar un proyecto en con Node:
npm init
Esto agrega el archivo package.json

* Agregar apartado script en el package.js:

"scripts": {
    "start": "node index.js",
    "dev":"nodemon index.js"
},

* Instalar Express en Node:

npm install express

Genera la carpeta node_modules y modifica las dependencias en package.json

* Instalar nodemon
npm install nodemon -D

* Instalar mongoose

npm install moongoose

Permite conectar a BD MongoDb

Configuracion de Conexion:

esto se deja en un archivo conexion.js

const mongoose = require("mongoose");
const urlMogoSrv = "mongodb://127.0.0.1:27017/tareas";

const connectDB = async () => {
  try {
    await mongoose.connect(urlMogoSrv, {
      useNewUrlParser: true,
      useUnifiedTopology: true,
    });
  } catch (error) {
    console.log(error);
  }
};

module.exports = connectDB;

luego se importa y se ejecuta desde index.js
const connectDB = require("./conexioDb");

connectDB();
